  totalTherapists: number;
  history: string[];
}
 
/**
 * Sidebar component displaying quick tips, roster stats, and recent activity
 * Provides helpful information and context for roster management
 */

  todayRoster, 
  totalTherapists, 
  history 
}: RosterSidebarProps) {

    ? Math.round((todayRoster.length / totalTherapists) * 100) 
    : 0;
 
  return (
    <div className="space-y-4">
      {/* Quick Tips Panel */}
      <div className="bg-gray-800 rounded-lg p-3">
        <h3 className="text-base font-semibold text-white mb-3 flex items-center gap-2">

          Quick Tips

            <span className="leading-tight">Use "Test Roster" for quick setup</span>
          </div>
          <div className="flex items-start gap-2">
            <span className="text-blue-400 text-xs">•</span>
            <span className="leading-tight">Click "All Clear" to start fresh</span>
          </div>
        </div>
      </div>
 
      {/* Roster Stats Panel */}
      <div className="bg-gray-800 rounded-lg p-3">
        <h3 className="text-base font-semibold text-white mb-3 flex items-center gap-2">
          <svg className="w-4 h-4 text-blue-400" fill="none" stroke="currentColor" viewBox="0 0 24 24">
            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9 19v-6a2 2 0 00-2-2H5a2 2 0 00-2 2v6a2 2 0 002 2h2a2 2 0 002-2zm0 0V9a2 2 0 012-2h2a2 2 0 012 2v10m-6 0a2 2 0 002 2h2a2 2 0 002-2m0 0V5a2 2 0 012-2h2a2 2 0 012 2v14a2 2 0 01-2 2h-2a2 2 0 01-2-2z" />

            <span className="text-gray-300 text-xs">Available:</span>
            <span className="text-green-400 font-semibold text-sm">{totalTherapists - todayRoster.length}</span>
          </div>

            <div className="text-xs text-gray-500 italic">No recent activity</div>
